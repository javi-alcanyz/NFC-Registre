#include <WiFi.h>
#include <HTTPClient.h>

#define BUTTON_PIN 15  // Pin donde está conectado el botón

const char* ssid = "TU_WIFI";
const char* password = "TU_CONTRASEÑA";
const char* serverName = "http://IP_DEL_SERVIDOR:5000/insertar";  // Cambia esto

void setup() {
  Serial.begin(115200);
  pinMode(BUTTON_PIN, INPUT_PULLDOWN);

  WiFi.begin(ssid, password);
  Serial.print("Conectando a WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nConectado!");
}

void loop() {
  if (digitalRead(BUTTON_PIN) == HIGH) {
    Serial.println("Botón presionado. Enviando datos...");

    if (WiFi.status() == WL_CONNECTED) {
      HTTPClient http;

      http.begin(serverName);
      http.addHeader("Content-Type", "application/json");

      String json = "{\"nombre\":\"Alumno Pico\",\"nfc_id\":\"boton_pico_001\"}";
      int httpResponseCode = http.POST(json);

      if (httpResponseCode > 0) {
        String response = http.getString();
        Serial.println("Respuesta del servidor: " + response);
      } else {
        Serial.print("Error en POST: ");
        Serial.println(httpResponseCode);
      }

      http.end();
    } else {
      Serial.println("WiFi desconectado.");
    }

    delay(1000);  // Espera para evitar múltiples envíos
  }
}
